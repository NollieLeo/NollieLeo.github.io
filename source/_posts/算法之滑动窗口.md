```js
/**
 * @param {number} target
 * @param {number[]} nums
 * @return {number}
 */
var minSubArrayLen = function (target, nums) {
    let left = 0;
    let right = 0;
    let minL = Infinity;
    let sum = 0;
    while (right < nums.length) {
        sum += nums[right];
        while (sum >= target) {
            minL = Math.min(minL, right - left + 1);
            sum -= nums[left];
            left++;
            if (sum < target) {
                break;
            }
        }
        right++;
    }
    return minL === Infinity ? 0 : minL
};

```
